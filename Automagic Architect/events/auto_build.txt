		name = auto_build.0001.a

	}

auto_build.0015 = {
	type = county_event
	theme = stewardship
	hidden = yes
	
	trigger = {
		NOT = { has_county_modifier = county_modifier_auto_build_disabled }
		holder = {
			OR = {
				has_character_flag = enable_auto_build
				AND = {
					has_character_flag = enable_vassal_auto_build
					is_vassal_of = root
				}
			}
		}
	}
	
	immediate = {
		if = {
			limit = {
				any_county_province = {
					has_holding_slot = yes
					NOT = { has_holding = yes }
				}
			}
			random_county_province = {
				limit = {
					has_holding_slot = yes
					NOT = { has_holding = yes }
				}
				save_scope_as = empty_holding_province
			}
			trigger_event = {
				id = auto_build.0016
				days = 1
			}
		}
	}
}

auto_build.0273 = {
	type = county_event
	theme = stewardship
	hidden = yes
	
	trigger = {
		NOT = { has_county_modifier = county_modifier_auto_build_disabled }
		holder = {
			OR = {
				has_character_flag = enable_auto_build
				AND = {
					has_character_flag = enable_vassal_auto_build
					is_vassal_of = event_target:auto_build_liege
				}
			}
		}
		any_county_province = {
			has_holding_slot = yes
			NOT = { has_holding = yes }
		}
	}
	
	weight_multiplier = {
		base = 100
		modifier = {
			factor = 10
			holder = {
				OR = {
					culture = { has_cultural_parameter = castle_holding }
					culture = { has_cultural_parameter = city_holding }
					culture = { has_cultural_parameter = church_holding }
				}
			}
		}
	}
	
	immediate = {
		random_county_province = {
			limit = {
				has_holding_slot = yes
				NOT = { has_holding = yes }
			}
			save_scope_as = empty_holding_province
		}
		
		holder = {
			if = {
				limit = {
					OR = {
						culture = { has_cultural_parameter = castle_holding }
						government_has_flag = government_is_feudal
					}
				}
				root = { trigger_event = auto_build.0015 }
			}
			else_if = {
				limit = {
					OR = {
						culture = { has_cultural_parameter = city_holding }
						government_has_flag = government_is_republic
					}
				}
				root = { trigger_event = auto_build.0015 }
			}
			else_if = {
				limit = {
					culture = { has_cultural_parameter = church_holding }
				}
				root = { trigger_event = auto_build.0015 }
			}
			else = {
				root = { trigger_event = auto_build.0015 }
			}
		}
	}
}

	}
