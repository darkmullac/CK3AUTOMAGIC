	desc = auto_build.0001.desc

		base = 1
	}
	override_background = {
		reference = terrain
	}
	left_portrait = {
		character = scope:root
	}
	immediate = {
		hidden_effect = {
			trigger_event = {
				id = auto_build.0003

		name = auto_build.0001.a

		name = auto_build.0001.b
