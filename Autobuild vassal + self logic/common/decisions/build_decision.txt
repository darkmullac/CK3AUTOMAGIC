my_auto_build_manager_decision = {
    picture = { reference = "gfx/interface/illustrations/holding_types/fp2_castle_iberian.dds" }


    desc = my_auto_build_manager_decision_desc
    is_shown = {
        is_ai = no
    }

    widget = {
        gui = "decision_view_widget_option_list_generic"
        controller = decision_option_list_controller
        decision_to_second_step_button = "my_auto_build_decision_second_step_button"
        show_from_start = yes

        # Activar auto_build normal
        item = {
            value = enable_normal_auto_build_item
            icon = "gfx/interface/icons/build_1.dds"
            localization = enable_auto_build
            is_valid = {
                NOT = {
                    has_character_flag = enable_auto_build
                    }
                    custom_tooltip = { text = auto_build_already_enabled }
                }
            }

        # Activar auto_build en vasallos
        item = {
            value = enable_vassal_auto_build_item
            icon = "gfx/interface/icons/build_2.dds"
            localization = enable_vassal_auto_build
            is_valid = {
                NOT = {
                    has_character_flag = enable_vassal_auto_build
                }
                custom_tooltip = { text = auto_build_vassal_already_enabled }
            }
        }

        # Desactivar auto_build
        item = {
            value = disable_auto_build_item
            icon = "gfx/interface/icons/build_3.dds"
            localization = disable_auto_build
            is_valid = {
                OR = {
                    has_character_flag = enable_vassal_auto_build
                    has_character_flag = enable_auto_build
                }
                custom_tooltip = { text = auto_build_not_enabled }
            }
        }
    }

    effect = {
        # Activar auto_build normal
        if = {
            limit = { scope:enable_normal_auto_build_item = yes }
            add_character_flag = enable_auto_build
            custom_tooltip = enable_auto_build_effect
            trigger_event = auto_build.0003
        }

        # Activar auto_build en vasallos
        else_if = {
            limit = { scope:enable_vassal_auto_build_item = yes }
            add_character_flag = enable_vassal_auto_build
            custom_tooltip = enable_vassal_auto_build_effect
            trigger_event = auto_build.0003
        }

        # Desactivar auto_build
        else_if = {
            limit = { scope:disable_auto_build_item = yes }
            remove_character_flag = enable_auto_build
            remove_character_flag = enable_vassal_auto_build
            custom_tooltip = disable_auto_build_effect
        }
    }

    decision_group_type = admin
    ai_check_interval = 0
    ai_potential = { always = no }
    ai_will_do = { base = 0 }
}
