	type = character_event
	trigger = {
		OR = {
			has_character_flag = enable_auto_build

	}

		reference = terrain

		character = scope:root
	}
	immediate = {
		hidden_effect = {
			trigger_event = {
				id = auto_build.0003
				days = 5
			}
		}
	}

		name = auto_build.0001.a
		add_character_flag = enable_auto_build
	}

		name = auto_build.0001.b
		add_character_flag = enable_vassal_auto_build
	}
